```c
long a = 10000, b = 0, c = 2800, d, e = 0, f[2801], g;

main() {
  for (; b - c;) f[b++] = a / 5;
  for (; d = 0, g = c * 2; c -= 14, printf("%.4d", e + d / a), e = d % a);
  for (b = c; d += f[b] * a, f[b] = d % --g, d /= g--, --b; d *= b);
}
```
